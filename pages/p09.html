custom alert

{selector}.alertUi({msg : '메시지입니다.',flag : 'confirm | ok', btn : '취소,확인'},function(){ /* 콜백함수 */});
flag 에서 컨펌,일반 얼럿 구분, btn에 버튼 이름 설정

이벤트 없이 바로 실행할 경우 :
$.fn.alertUi('',{msg : '메시지입니다.',flag : 'confirm | ok', btn : '취소,확인'},function(){ /* 콜백함수 */});
