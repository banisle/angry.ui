custom alert

타이틀 있을 때 :
ANUI.module.alertUi({title : 'tit',msg : 'aaa'},function(){ /* 콜백함수 */});
타이틀 있을 때 :
ANUI.module.alertUi({msg : 'aaa'},function(){ /* 콜백함수 */});
타이틀 있을 때, 없을 때 모두 alertui 호출
타이틀이 있으면 title 값 입력, 없으면 입력X 

<button class="button" id="aa">alert 창 - 타이틀 있는 것</button>
<button class="button" id="bb">alert 창 - 타이틀 없는 것</button>

<script>
	var _call_back = function(){
		//callback 내용
		console.log(1);
	}
	$('#aa').on('click', function(){//#aa를 눌렀을 때
		ANUI.module.alertUi({title : 'tit',msg : 'aaa'},function(){
			_call_back();
		});
	})
	$('#aa').on('click', function(){//#bb를 눌렀을 때
		ANUI.module.alertUi({msg : 'aaa'},function(){
			_call_back();
		});
	})
</script>

toast

ANUI.module.toastUi({msg:'메시지 내용용, time: 토스트 사라지는 시간}, callback)

$('#toastCall').on('click', function(){
    ANUI.module.toastUi({
        msg: '토스트 메시지 출력력',
        time: 1000
    }, function(){
        _call_back();
    });
});
