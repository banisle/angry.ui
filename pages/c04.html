ajax 모달 

$("#ajax_btn").on('click', function (e) {
    
    e.preventDefault();
    var url = this.href,
        dataVal = $('#ajax_input').val(),
        //버튼에 넘겨주기 위한 변수
        $t = $(this);

    $.ajax({
        type: "get",
        url: url,
        dataType: "html",
        success: function () {
            $("#ajax_data").text(dataVal);
            // 현재 버튼을 $t 인자값으로 넘겨줌
            $('#a').modalUi('open', 'ajax_modal01', true, $t);

        },
        error: function (request, status, error) {
            console.log("code:" + reque가t.status + "\n" + "message:" + request.responseText +
                "\n" + "error:" + error);
        }
    });
});

$('.mvClose').on('click', function () {
    $('#a').modalUi('close', 'ajax_modal01');
})
