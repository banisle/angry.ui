css var color change

input box 에 변경하고자 하는 color 값 적용(hex,rgba 상관없음) 

<div style="margin-top:10px; width:100%;height:50px;background-color: var(--main-clr);"></div>

<style>
	:root{
		--main-clr: red;
	}
</style>

<script>
const $chgBtn = document.getElementById('chgBtn');

function chgColor(){
	const _chgClrInp = document.getElementById('chgClrInp').value;

	//메인 변수 컬러 가져오기
	getComputedStyle(document.documentElement).getPropertyValue('--main-clr');

	//메인 변수 컬러 지정
	document.documentElement.style.setProperty('--main-clr', _chgClrInp);
}

$chgBtn.addEventListener('click',function(){
	chgColor();
});
</script>
