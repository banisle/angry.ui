TreeSixtyImageRotate

이미지 특정(1) 프레임에 제어용 클래스를 추가, 해당 프레임에 클래스 액티브 적용

icon mark
<div class="ui-360rot">
    <div class="rotateWrap">
        <div class="icn">icon mark</div>
        <div class="threesixty-image-rotate product"></div>
    </div>
</div>

<script>
    $(document).ready(function () {
        $('.product').TreeSixtyImageRotate({
            totalFrames: 36,
            endFrame: 36,
            currentFrame: 0,
            extension: ".png",
            imagesFolder: "../img/prod1/",
            smallWidth: 400,
            smallHeight: 400,
            largeWidth: 800,
            largeHeight: 800,
            imagePlaceholderClass: "images-placeholder"
        }).initTreeSixty();

        var icnActive = function () {

            var $this = $('.rotateWrap'),
                //이미지 첫번째 프레임
                $img0 = $this.find('.image-0'),
                isActive,
                $icn = $this.find('.icn'),
                $playBtn = $this.find('.navigation-bar-play');


            $this.on('mouseover', function () {
                var isActive = $img0.hasClass('active');
                isActive ? $icn.addClass('active') : $icn.removeClass('active');

                console.log(isActive);
            });

            $playBtn.on('click', function () {
                $this.trigger('mouseover');
            })

        };

        icnActive();
    });
</script>
